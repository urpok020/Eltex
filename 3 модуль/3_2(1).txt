#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/msg.h>

#define MSG_SIZE 128
#define MSG_TYPE 1

struct msgbuf {
    long mtype;
    char mtext[MSG_SIZE];
};

int main() {
    key_t key;
    int msgid;
    struct msgbuf message;

    key = ftok("/tmp", 'A');
    msgid = msgget(key, 0666 | IPC_CREAT);

    message.mtype = MSG_TYPE;
    strcpy(message.mtext, "Hello, this is the sender process");

    msgsnd(msgid, &message, sizeof(message.mtext), 0);

    // Отправляем сообщение с типом 255 для завершения обмена
    message.mtype = 255;
    strcpy(message.mtext, "Finish");

    msgsnd(msgid, &message, sizeof(message.mtext), 0);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Contact {
    char name[50];
    char job[50];
    char phone[15];
    char email[50];
    char social[50];
    struct Contact* next;
    struct Contact* prev;
};

struct Contact* head = NULL;
struct Contact* tail = NULL;

void addContact(char name[], char job[], char phone[], char email[], char social[]) {
    struct Contact* newContact = (struct Contact*)malloc(sizeof(struct Contact));
    
    strcpy(newContact->name, name);
    strcpy(newContact->job, job);
    strcpy(newContact->phone, phone);
    strcpy(newContact->email, email);
    strcpy(newContact->social, social);
    
    newContact->next = NULL;
    
    if (head == NULL) {
        head = newContact;
        tail = newContact;
        newContact->prev = NULL;
    } else {
        tail->next = newContact;
        newContact->prev = tail;
        tail = newContact;
    }
}

void displayContacts() {
    struct Contact* current = head;
    
    if (head == NULL) {
        printf("Phonebook is empty\n");
    } else {
        while (current != NULL) {
            printf("Name: %s\n", current->name);
            printf("Job: %s\n", current->job);
            printf("Phone: %s\n", current->phone);
            printf("Email: %s\n", current->email);
            printf("Social: %s\n", current->social);
            printf("\n");
            current = current->next;
        }
    }
}
int main() {
    int choice;
    
    do {
        printf("Phonebook Menu\n");
        printf("1. Add Contact\n");
        printf("2. Display Contacts\n");
        printf("3. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);
        
        switch (choice) {
            case 1: {
                char name[50];
                char job[50];
                char phone[15];
                char email[50];
                char social[50];
                
                printf("Enter name: ");
                scanf("%s", name);
                printf("Enter job: ");
                scanf("%s", job);
                printf("Enter phone number: ");
                scanf("%s", phone);
                printf("Enter email: ");
                scanf("%s", email);
                printf("Enter social media: ");
                scanf("%s", social);
                
                addContact(name, job, phone, email, social);
                printf("Contact added successfully\n");
                break;
            }
            case 2: {
                displayContacts();
                break;
            }
            case 3: {
                printf("Exiting...\n");
                break;
            }
            default: {
                printf("Invalid choice\n");
                break;
            }
        }
    } while (choice != 3);
    
    return 0;
}
